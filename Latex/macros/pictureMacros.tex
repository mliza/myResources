\newcommand{\pic}[4]{
\begin{figure}[H]
\centering
\includegraphics[scale=#1]{#2}
\caption{#3} \label{#4}
\end{figure}		}

    \begin{figure}[H]
        \centering
        \begin{minipage}{0.49\textwidth}
            \label{isometric}

                \caption{#2} \label{#3}
        \end{minipage}
        \hfill
        \begin{minipage}{0.49\textwidth}
            \label{side}
            \includegraphics[width=\textwidth]{#4}
            \caption{#5} \label{#6}
        \end{minipage} 
    \end{figure}         }
